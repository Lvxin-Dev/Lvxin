CREATE TABLE IF NOT EXISTS public.companies
(
    comp_id text COLLATE pg_catalog."default" NOT NULL,
    password text COLLATE pg_catalog."default" NOT NULL,
    email text COLLATE pg_catalog."default" NOT NULL,
    username text COLLATE pg_catalog."default" NOT NULL,
    license text COLLATE pg_catalog."default" NOT NULL,
    hotline text NOT NULL,
    subscription text,
    CONSTRAINT comp_pkey PRIMARY KEY (comp_id),
    CONSTRAINT comp_email_key UNIQUE (email),
    CONSTRAINT comp_username_key UNIQUE (username)
)
